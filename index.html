// Recibir props SIN desestructurar (evitar):

✅ Solución 
import { useState } from "react"

// Extendemos PokemonCard del ejercicio 1 añadiendo la prop stats
const PokemonCard = ({ name, number, image, types, stats }) => {
  const [verStats, setVerStats] = useState(false)

  return (
    <div className="poke-card">
      <img src={image} alt={name} />
      <h3>{name}</h3>
      <p className="num">#{number}</p>
      <div className="poke-types">
        {types.map(t => <span key={t} className={`poke-type t-${t}`}>{t}</span>)}
      </div>

      <button className="toggle-btn" onClick={() => setVerStats(!verStats)}>
        {verStats ? "Ocultar stats" : "Ver stats"}
      </button>

      {verStats && (
        <div className="poke-stats">
          {stats.map(({ name: statName, value }) => (
            <div key={statName} className="stat-row">
              <span className="stat-name">{statName}</span>
              <div className="stat-bar-bg">
                <div className="stat-bar"
                  style={{ width: `${(value / 255) * 100}%` }}
                />
              </div>
              <span className="stat-val">{value}</span>
            </div>
          ))}
        </div>
      )}
    </div>
  )
}

⚙️ Crea useAuth.js en src/hooks/ antes de empezar.

La chuleta muestra un patrón de autenticación con useAuth(). Tu tarea es implementarlo: si el usuario está autenticado se muestra el Dashboard, si no se muestra la pantalla de Login. Un botón permite alternar entre los dos estados.

Lo que tienes que implementar

  2. Crea el componente Login: muestra un mensaje y un botón "Iniciar sesión" que llama a onLogin (prop).
  3. Crea el componente Dashboard: muestra un mensaje de bienvenida y un botón "Cerrar sesión" que llama a onLogout (prop).
  4. En App, llama a useAuth() y usa renderizado condicional para mostrar Login o Dashboard según isAuth.

function App() {
  const isAuthUser = useAuth()

💡 Pista
El Custom Hook devuelve dos cosas: el booleano y la función para cambiarlo. En App puedes desestructurarlos: const {'{ isAuth, toggleAuth }'} = useAuth(). Luego pasa toggleAuth como prop a Login y Dashboard. Para el condicional puedes usar if antes del return o ternario dentro del JSX.
✅ Solución
useAuth.js
import { useState } from "react"

function useAuth() {
  const [isAuth, setIsAuth] = useState(false)
  const toggleAuth = () => setIsAuth(!isAuth)
  return { isAuth, toggleAuth }
}

export default useAuth
App.jsx
import useAuth from "./useAuth"

const Login = ({ onLogin }) => (
  <div className="auth-screen">
    <h2>🔒 Pokédex</h2>
    <p>Inicia sesión para ver los Pokémon</p>
    <button className="auth-btn login" onClick={onLogin}>
      Iniciar sesión
    </button>
  </div>
)

const Dashboard = ({ onLogout }) => (
  <div className="dashboard">
    <h2>✅ Bienvenido, Entrenador</h2>
    <p>Ya puedes ver tu Pokédex</p>
    <button className="auth-btn logout" onClick={onLogout}>

  return <Dashboard onLogout={toggleAuth} />
}

✅ Solución 
import { useState } from "react"

const pokemons = [
  { id: 25,  name: "pikachu",   number: "025" },
  { id: 1,   name: "bulbasaur",  number: "001" },
  { id: 4,   name: "charmander", number: "004" },
  { id: 7,   name: "squirtle",   number: "007" },
  { id: 39,  name: "jigglypuff", number: "039" },
]

const App = () => {
  const [indice, setIndice] = useState(0)
  const { id, name, number } = pokemons[indice]

  return (
    <>
      <div className="poke-card">
        <img src={`.../${id}.png`} alt={name} />
        <h3>{name}</h3>
        <p>#{number}</p>
      </div>
      <div className="poke-nav">
        <button
          disabled={indice === 0}
          onClick={() => setIndice(indice - 1)}
        >← Anterior</button>
        <button
          disabled={indice === pokemons.length - 1}
          onClick={() => setIndice(indice + 1)}
        >Siguiente →</button>
      </div>
      <p className="poke-counter">{indice + 1} / {pokemons.length}</p>
    </>
  )
}

import { useState, useEffect } from "react"

function usePokedex() {
  const [pokemons, setPokemons] = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    const fetchAll = async () => {
      try {
        const res = await fetch("https://pokeapi.co/api/v2/pokemon?limit=9")
        const data = await res.json()
        // Peticiones en paralelo para cada Pokémon
        const detalles = await Promise.all(
          data.results.map(p => fetch(p.url).then(r => r.json()))
        )
        setPokemons(detalles.map(d => ({
          id:    d.id,
          name:  d.name,
          image: d.sprites.other["official-artwork"].front_default,
          types: d.types.map(t => t.type.name)
        })))
      } catch(err) {
        setError(err.message)
      } finally {
        setLoading(false)
      }
    }
    fetchAll()
  }, [])

  return { pokemons, loading, error }
}
